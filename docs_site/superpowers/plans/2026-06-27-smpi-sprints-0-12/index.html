    SMPI Sprints 0–12 Implementation Plan¶

    For agentic workers: Use superpowers:subagent-driven-development to implement task-by-task.

    Goal: Implement the full SMPI system from Sprint 0 (foundation) through Sprint 12 (Docker Swarm deploy), covering all 13 sprints defined in PRD.md §20.

    Architecture: Single-tenant Django 6 app with DRF, Celery/RabbitMQ/Redis, PostgreSQL+pgvector, LangGraph prescriptive AI pipeline, SSE chat stream, and Docker Swarm deploy with Traefik TLS.

    Tech Stack: Python 3.13 · Django 6 · DRF · Celery · LangGraph · sentence-transformers · pgvector · scikit-learn · Reportlab · Evolution API · Docker Swarm · Traefik

    Global Constraints¶

    • Python ≥ 3.13, Django ≥ 6.0
    • No automated tests (explicitly excluded — PRD §4.2)
    • All models inherit TimeStampedModel (created_at, updated_at)
    • Login by email only (no username)
    • AUTH_USER_MODEL = 'accounts.User'
    • Feature vector: 18 metric columns (mm/s, °C, g, hz — excluding redundant in/s and °F)
    • Embedding dimension: 384 (paraphrase-multilingual-MiniLM-L12-v2)
    • States: normal, baseline, teste, acelerando, motor_desligado → no prescription
    • Guard rule: no docs → report undocumented, never invent solution
    • All tasks heavy → Celery async, notify via notifications app on complete
    • Design system: FIESC tokens (--fsi-clr-, --fs-, MuseoSans), VLibras on all pages
    • Agent to use: django-backend-smpi (backend), django-frontend-fiesc (templates), smpi-ai-pipeline (AI)

class ApiKeyAuthentication(BaseAuthentication):
    def authenticate(self, request):
        key = request.META.get('HTTP_X_API_KEY') or request.GET.get('api_key')
        if not key:
            return None
        try:
            api_key = ApiKey.objects.select_related('user').get(key=key, is_active=True)
        except ApiKey.DoesNotExist:
            raise AuthenticationFailed('Invalid or inactive API key.')
        api_key.last_used_at = timezone.now()
        api_key.save(update_fields=['last_used_at'])
        return (api_key.user, api_key)

import joblib
from pathlib import Path
from django.conf import settings
from .models import FEATURE_COLUMNS

def extract_features(metrics: dict) -> list[float]:
    """Extract and return ordered feature list from raw metrics dict."""
    return [float(metrics.get(col, 0.0) or 0.0) for col in FEATURE_COLUMNS]

def fit_and_save_scaler(feature_matrix: np.ndarray) -> object:
    from sklearn.preprocessing import StandardScaler
    scaler = StandardScaler()
    scaler.fit(feature_matrix)
    path = Path(settings.SCALER_PATH)
    path.parent.mkdir(parents=True, exist_ok=True)
    joblib.dump(scaler, path)
    return scaler

def load_scaler():
    path = Path(settings.SCALER_PATH)
    if not path.exists():
        return None
    return joblib.load(path)

def normalize_features(raw_features: list[float]) -> list[float]:
    scaler = load_scaler()
    if scaler is None:
        return raw_features
    arr = np.array(raw_features).reshape(1, -1)
    return scaler.transform(arr)[0].tolist()

      from pgvector.django import L2Distance
from .models import SensorReading, FEATURE_DIM
from .preprocessing import normalize_features, extract_features

def find_similar_readings(feature_vector: list[float], k: int = 10, exclude_id: int = None):
    """Return K nearest SensorReadings that are problems, ordered by L2 distance."""
    from pgvector.django import VectorField
    import numpy as np
    qs = SensorReading.objects.filter(
        status_class='problem',
        feature_vector__isnull=False,
    )
    if exclude_id:
        qs = qs.exclude(id=exclude_id)
    vec = np.array(feature_vector)
    return qs.annotate(distance=L2Distance('feature_vector', vec)).order_by('distance')[:k]

def identify_fault_by_voting(similar_readings):
    """Weighted vote by 1/distance to pick most likely fault code."""
    from collections import defaultdict
    votes = defaultdict(float)
    for r in similar_readings:
        if r.fault_id and r.distance and r.distance > 0:
            votes[r.fault_id] += 1.0 / r.distance
    if not votes:
        return None

def execute_pipeline(reading: SensorReading) -> Prescription:
    graph = build_graph()
    initial_state = {
        'reading_id': reading.id,
        'metrics': reading.metrics,
        'feature_vector': [],
        'status_class': 'pending',
        'similar_readings': [],
        'fault_id': None,

        'error': None,

    p, _ = Prescription.objects.update_or_create(

            'defect_type': result.get('fault_code', ''),
            'occurrences_count': result.get('occurrences_count', 0),
            'occurrences_frequency': result.get('occurrences_frequency', ''),
